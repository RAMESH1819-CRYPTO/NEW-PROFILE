# Python Voice Assistant Implementation import speech_recognition as sr import pyttsx3 import datetime # Initialize speech engine engine = pyttsx3.init() def speak(text): """Convert text to speech""" engine.say(text) engine.runAndWait() def listen(): """Listen to user voice input""" recognizer = sr.Recognizer() with sr.Microphone() as source: print("Listening...") audio = recognizer.listen(source) try: text = recognizer.recognize_google(audio) return text.lower() except: return "" def process_command(command): """Process voice commands""" if "weather" in command: speak("Checking weather...") elif "time" in command: time = datetime.datetime.now().strftime("%H:%M") speak(f"The time is {time}") elif "search" in command: speak("Searching the web...") # Main loop while True: command = listen() if command: process_command(command)
